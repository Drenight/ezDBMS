SELECT 
rela_i_i_1000.key, MIN(rela_i_i_1000.val), AVG(rela_i_i_100000.key), SUM(rela_i_i_100000.val)

FROM 
rela_i_i_1000 

INNER JOIN 
rela_i_i_100000 

ON 
rela_i_i_1000.key = rela_i_i_100000.key 

WHERE
rela_i_i_1000.key>5 OR rela_i_i_1000.key!=1000

GROUP BY 
rela_i_i_1000.key 

HAVING 
MIN(rela_i_i_1000.val)>=20 OR AVG(rela_i_i_100000.val)>550

ORDER BY 
rela_i_i_1000.key ASC

LIMIT 
100;
